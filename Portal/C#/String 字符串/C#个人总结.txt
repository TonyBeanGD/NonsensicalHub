
namespace Luck_draw
{
    class Program
    {
        static void Main(string[] args)
        {
            string[] n_p;
            string memeberName = string.Empty;
            for(;;)
            {
                Console.WriteLine("请输入一个人的名字,如果没有更多的人请直接输入回车");
                string temp = Console.ReadLine();
                if (temp == "") break;
                memeberName += temp + '&';
            }
            //去掉最尾部的'&'
            memeberName = memeberName.TrimEnd('&');
            //将字符串根据'&'字符分成多个字符串并赋值给n_p字符串数组
            n_p = memeberName.Split('&');
            Draw get = new Draw();
            int l=get.draw(n_p);
            Console.WriteLine("中奖者为{0}",n_p[l]);
            Console.ReadKey();
        }
    }
}
