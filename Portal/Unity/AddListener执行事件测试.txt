//unity3d，5.6.2
    //往button添加监听
    //代码
    private void Start()
    {
        for (int i = 0; i < 10; i++)
        {
            int j = i;
            transform.GetChild(i).GetComponent<Button>().onClick.AddListener(delegate 
            {
                int k = i;
                Debug.Log(i);
                Debug.Log(j);
                Debug.Log(k);
            }) ;
        }
    }
    //结果：i和k的值一直为10，j的值不同
    //猜测：AddListener方法不会马上执行，应该是在每一帧靠后的时间段里面执行
