namespace DataEncryptDecrypt
{
    public class EncryptDecrypt
    {
        private static string HEXADECIMAL = "0123456789abcdef";

        /// <summary>
        /// 数据解密
        /// </summary>
        /// <param name="jsonData">待解密string</param>
        /// <returns>解密出的json对象</returns>
        public static JsonData dataDecrypt(string jsonData)
        {
            try
            {
                JsonData jd = JsonMapper.ToObject(jsonData);

                string formData = jd["data"].ToString();

                JsonData parentObject = JsonMapper.ToObject(System.Text.Encoding.UTF8.GetString(Convert.FromBase64String(formData)));

                string uuId = parentObject["uuId"].ToString();
                string oldUUID = uuIdRight(uuId);

                string data2 = parentObject["data2"].ToString();

                JsonData dataObject2 = JsonMapper.ToObject(AesDecrypt(data2, oldUUID));

                string data1 = parentObject["data1"].ToString();
                string resultJson = AesDecrypt(AesDecrypt(AesDecrypt(data1, dataObject2["5"].ToString()), dataObject2["2"].ToString()), dataObject2["6"].ToString());

                return JsonMapper.ToObject(resultJson);
            }
            catch (System.Exception e)
            {
                Console.WriteLine(e.Message);
                return null;
            }
        }

        /// <summary>
        /// 数据加密
        /// </summary>
        /// <param name="jsonData">待加密string</param>
        /// <returns>加密出的string密文</returns>
        public static string dataEncrypt(string jsonData)
        {
            Dictionary<string, string> resultData = new Dictionary<string, string>();
            try
            {
                Dictionary<string, string> mapData2 = new Dictionary<string, string>();
                string uuId = Guid.NewGuid().ToString().Replace("-", "");

                mapData2.Add("1", StrToMD5(StrToMD5(StrToMD5(uuId))));
                mapData2.Add("2", StrToMD5(StrToMD5(StrToMD5(StrToMD5(StrToMD5(StrToMD5(uuId)))))));
                mapData2.Add("3", StrToMD5(StrToMD5(StrToMD5(StrToMD5(uuId)))));
                mapData2.Add("4", StrToMD5(uuId));
                mapData2.Add("5", StrToMD5(StrToMD5(uuId)));
                mapData2.Add("6", StrToMD5(StrToMD5(StrToMD5(StrToMD5(StrToMD5(uuId))))));

                string dataAES = AesEncrypt(AesEncrypt(AesEncrypt(jsonData, mapData2["6"]), mapData2["2"]), mapData2["5"]);
                string jsonData2 = SerializeDictionaryToJsonString(mapData2);
                string dataAES2 = AesEncrypt(jsonData2, uuId);

                string uuIdNew = uuIdLeft(uuId);
                resultData.Add("data1", dataAES);
                resultData.Add("data2", dataAES2);
                resultData.Add("uuId", uuIdNew);
                string temp = Convert.ToBase64String(System.Text.Encoding.UTF8.GetBytes(SerializeDictionaryToJsonString(resultData)), Base64FormattingOptions.None);

                return temp;
            }
            catch (System.Exception e)
            {
                Console.WriteLine(e.Message);
                return null;
            }
        }

        private static string StrToMD5(string str)
        {
            byte[] data = Encoding.UTF8.GetBytes(str);
            MD5 md5 = new MD5CryptoServiceProvider();
            byte[] OutBytes = md5.ComputeHash(data);

            string OutString = "";
            for (int i = 0; i < OutBytes.Length; i++)
            {
                OutString += OutBytes[i].ToString("x2");
            }

            return OutString.ToLower();
        }

        /// <summary>
        ///  AES 加密
        /// </summary>
        /// <param name="str">明文（待加密）</param>
        /// <param name="key">密文</param>
        /// <returns></returns>
        private static string AesEncrypt(string str, string key)
        {
            if (string.IsNullOrEmpty(str)) return null;
            byte[] toEncryptArray = Encoding.UTF8.GetBytes(str);

            RijndaelManaged rm = new RijndaelManaged
            {
                Key = Encoding.UTF8.GetBytes(key),
                Mode = CipherMode.ECB,
                Padding = PaddingMode.PKCS7
            };

            ICryptoTransform cTransform = rm.CreateEncryptor();
            byte[] resultArray = cTransform.TransformFinalBlock(toEncryptArray, 0, toEncryptArray.Length);

            return Convert.ToBase64String(resultArray, 0, resultArray.Length);
        }

        /// <summary>
        ///  AES 解密
        /// </summary>
        /// <param name="str">明文（待解密）</param>
        /// <param name="key">密文</param>
        /// <returns></returns>
        private static string AesDecrypt(string str, string key)
        {
            if (string.IsNullOrEmpty(str)) return null;
            byte[] toEncryptArray = Convert.FromBase64String(str);

            RijndaelManaged rm = new RijndaelManaged
            {
                Key = Encoding.UTF8.GetBytes(key),
                Mode = CipherMode.ECB,
                Padding = PaddingMode.PKCS7
            };

            ICryptoTransform cTransform = rm.CreateDecryptor();
            byte[] resultArray = cTransform.TransformFinalBlock(toEncryptArray, 0, toEncryptArray.Length);

            return Encoding.UTF8.GetString(resultArray);
        }

        /// <summary>
        /// 将字典类型序列化为json字符串
        /// </summary>
        /// <typeparam name="TKey">字典key</typeparam>
        /// <typeparam name="TValue">字典value</typeparam>
        /// <param name="dict">要序列化的字典数据</param>
        /// <returns>json字符串</returns>
        private static string SerializeDictionaryToJsonString<TKey, TValue>(Dictionary<TKey, TValue> dict)
        {
            if (dict.Count == 0)
                return "";

            return JsonMapper.ToJson(dict);
        }

        private static string uuIdLeft(string uuId)
        {
            var s = HEXADECIMAL.IndexOf(uuId[16]);
            string newIndex16 = HEXADECIMAL[s - 1].ToString();
            return uuIdReplace(uuId, newIndex16);
        }

        private static string uuIdRight(string uuId)
        {
            var s = HEXADECIMAL.IndexOf(uuId[16]);
            string newIndex16 = HEXADECIMAL[s + 1].ToString();
            return uuIdReplace(uuId, newIndex16);
        }

        private static string uuIdReplace(string uuId, string str)
        {
            StringBuilder sb = new StringBuilder(uuId);
            sb[16] = str[0];
            return sb.ToString();
        }
    }

}
